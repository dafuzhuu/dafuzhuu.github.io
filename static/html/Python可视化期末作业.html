前言¶

本文档是《Python数据获取、处理和可视化》的课程作业，选题是“大型量化私募管理规模的地域分布统计”。这段时间在找量化私募的实习，之前进小私募实习吃了点亏，所以我比较看重其管理规模。一般来讲，小私募管理不太规范，路子比较野；大私募无论管理、风控都做的更好。但多小算小，多大算大呢？回答了这个问题以后，大私募在地域上又呈现怎样的分布呢？

于是我想到从证券基金业协会上将私募机构信息用爬虫抓取下来，先进行描述性统计，然后按地域做可视化。

数据获取¶

这个网站反爬机制还挺难处理的，刚登陆进私募基金管理人分类查询公示页面时会跳出五秒的提示窗口，直接用 requests 爬不到被遮住的信息。后来尝试了在 requests.get 中加入 cookie，但还是失败。最后的解决办法是用 selenium 模拟操作浏览器，硬性等待5秒后刷新，让窗口消失，然后把每个私募的链接抓取下来。然后遍历全部链接，用 requests 抓取每个链接内的单个私募信息，总共24718条，最后以 DataFrame 的格式汇总在一起，输出为 private_equity_info.xlsx

如果请求太频繁会被网站中断，报错信息：

ConnectionError: ('connection aborted.'，connectionResetError(18854，'远程主机强迫关闭了一个现有的连接。'，None，18854，None))

因此遍历全部链接时要分批处理

class FundScraper:
    def __init__(self, url, batch_size=100, retry_attempts=3):
        self.url = url
        self.batch_size = batch_size
        self.retry_attempts = retry_attempts
        self.links = []
        self.info_list = []
        self.failed_links = []
        self.driver = webdriver.Chrome()

    # 获取当前页面的链接
    def get_links(self):
        self.driver.get(self.url)
        time.sleep(5.1)
        self.driver.refresh()

        while True:
            rows = WebDriverWait(self.driver, 10).until(
                EC.presence_of_all_elements_located((By.XPATH, '//td[@data-dt-column="1"]'))
            )
            for row in rows:
                try:
                    a_tag = row.find_element(By.TAG_NAME, 'a')
                    href = a_tag.get_attribute('href')
                    self.links.append(href)
                except:
                    print("No <a> tag found")
            
            # 检查是否需要结束循环
            try:
                disabled_element = self.driver.find_element(By.XPATH, '//a[@disabled="disabled"]')
                if disabled_element and disabled_element.text == '1028':  # 最后一页
                    print("Last page")
                    break
            except:
                pass
            
            # 查找“下一页”按钮并点击
            try:
                next_button = WebDriverWait(self.driver, 10).until(
                    EC.element_to_be_clickable((By.XPATH, '//a[@class="paginate_button next"]'))
                )
                self.driver.execute_script("arguments[0].scrollIntoView();", next_button)
                actions = ActionChains(self.driver)
                actions.move_to_element(next_button).click().perform()
                time.sleep(0.3)
            except Exception as e:
                print(f"An error occurred: {e}")
                break

        self.driver.quit()

    @staticmethod
    def parse_html(response):
        try:
            soup = BeautifulSoup(response.content, 'html.parser')
            fund_info = {
                "基金管理人全称(中文)": None,
                "机构类型": None,
                "办公地址": None,
                "管理规模区间": None,
                "全职员工人数": None
            }

            element = soup.find('span', id='complaint2')
            if element:
                fund_info["基金管理人全称(中文)"] = element.text.strip()

            table_rows = soup.select('table.table tbody tr')
            for row in table_rows:
                cells = row.find_all('td')
                if len(cells) > 0:
                    cell_title = cells[0].get_text(strip=True)
                    cell_value = cells[1].get_text(strip=True) if len(cells) > 1 else None
                    if cell_title == "办公地址":
                        fund_info["办公地址"] = cell_value
                    elif cell_title == "管理规模区间":
                        fund_info["管理规模区间"] = cell_value
                    elif cell_title == "全职员工人数":
                        fund_info["全职员工人数"] = cell_value
                    elif cell_title == "机构类型":
                        fund_info["机构类型"] = cell_value

            return fund_info, None
        except Exception as e:
            return None, str(e)

    # 定义获取信息的函数
    def fetch_info(self, link):
        try:
            response = requests.get(link, timeout=10)
            response.raise_for_status()
            parsed_info, error = self.parse_html(response)
            return link, parsed_info, error
        except requests.RequestException as e:
            return link, None, str(e)

    def batch_links(self, links):
        for i in range(0, len(links), self.batch_size):
            yield links[i:i + self.batch_size]

    # 访问太频繁会被踢出去，将链接列表分批
    def process_links(self):
        for batch in tqdm(self.batch_links(self.links), total=len(self.links)//self.batch_size, desc='Batches'):
            with concurrent.futures.ThreadPoolExecutor() as executor:
                results = list(tqdm(executor.map(self.fetch_info, batch), total=len(batch), desc='urls'))
            for link, info, error in results:
                if info is not None:
                    self.info_list.append(info)
                else:
                    self.failed_links.append(link)
    
    # 处理失败的链接（可以选择重新尝试几次）
    def retry_failed_links(self):
        for attempt in range(self.retry_attempts):
            if not self.failed_links:
                break
            current_failed_links = self.failed_links
            self.failed_links = []
            for batch in tqdm(self.batch_links(current_failed_links), total=len(current_failed_links)//self.batch_size, desc=f'Retrying attempt {attempt + 1}'):
                with concurrent.futures.ThreadPoolExecutor() as executor:
                    results = list(tqdm(executor.map(self.fetch_info, batch), total=len(batch), desc='urls'))
                for link, info, error in results:
                    if info is not None:
                        self.info_list.append(info)
                    else:
                        self.failed_links.append(link)

    def run(self):
        self.get_links()
        self.process_links()
        self.retry_failed_links()
        return self.info_list

数据预处理¶

由于私募基金分为证券基金、股权基金等，股权基金从事的是风险投资相关业务，而量化私募指的是证券基金，我们要将其筛选出来。

另外，我们还需从办公地址中提取其所在的第一行政区划（省/直辖市/自治区），基金业协会注册的机构中不含香港、澳门、台湾的机构。

针对以上两点，我们对原始数据进行预处理。

描述性统计¶

我们首先观察私募数量在规模区间上的分布。

将所有私募规模都向下取值，比如“10-20亿元”记为10亿元，这样做可以避免超大型私募对整体的统计结果造成较大偏差。观察分位数线，我们50%以上的记为中型私募，暂不考虑百亿以上的大型私募。

最后，虽然在职员工人数跟私募大小没有直接关系（往往管理百亿资金的团队只有十几人），但我们也探究下员工人数呈现怎样的分布。

class FundScaleAnalysis:
    def __init__(self, data):
        self.data = data
        self.grouped = None

    def group_data(self):
        self.grouped = self.data["管理规模区间"].value_counts().reset_index()
        self.grouped.columns = ['size', 'count']

    # 管理规模柱状图
    def plot_bar_chart(self):
        plt.figure(figsize=(12, 2), dpi=500)
        plt.style.use('ggplot')
        matplotlib.rcParams['font.sans-serif'] = ['KaiTi']
        bars = plt.bar(self.grouped['size'], self.grouped['count'], color='skyblue')
        plt.title('管理规模分布图', fontsize=16)
        plt.xlabel('规模', fontsize=14)
        plt.ylabel('数量', fontsize=14)

        for bar in bars:
            yval = bar.get_height()
            plt.text(bar.get_x() + bar.get_width()/2 - 0.1, yval + 0.5, int(yval), fontsize=12)

        plt.xticks(rotation=30)
        plt.show()

    def size_boxplot_data(self):
        data = {
            '规模': self.grouped["size"].values.tolist(),
            '数量': self.grouped["count"].values.tolist()
        }
        df = pd.DataFrame(data)
        df['规模数值'] = df['规模'].apply(self.scale_to_numeric)
        return df

    @staticmethod
    def scale_to_numeric(scale):
        if '亿元以上' in scale:
            return 100
        return int(scale.split('-')[0].replace('亿元', ''))

    def plot_boxplot(self, df, column, title):
        plt.figure(figsize=(10, 2), dpi=300)
        plt.style.use('ggplot')
        plt.boxplot(df[column], vert=False, patch_artist=True, boxprops=dict(facecolor='skyblue'))
        plt.xlabel('数量', fontsize=14)
        plt.yticks([1], [title])
        plt.show()

    def describe_statistics(self, df, column):
        desc_stats = df[column].describe()
        return pd.DataFrame(desc_stats)

    def run_analysis(self):
        self.group_data()
        self.plot_bar_chart()
        df = self.size_boxplot_data()
        self.plot_boxplot(df, '规模数值', '规模')
        desc_stats_scale = self.describe_statistics(df, '规模数值')
        print(desc_stats_scale)

        # 对在职员工人数进行分析
        if "全职员工人数" in self.data.columns:
            df_employees = self.data.dropna(subset=["全职员工人数"]).copy()
            df_employees["全职员工人数"] = pd.to_numeric(df_employees["全职员工人数"], errors='coerce')
            df_employees = df_employees.dropna(subset=["全职员工人数"])
            self.plot_boxplot(df_employees, '全职员工人数', '全职员工人数')
            desc_stats_employees = self.describe_statistics(df_employees, '全职员工人数')
            print(desc_stats_employees)

class ChartGenerator:
    def __init__(self, data, title):
        self.data = data
        self.title = title

    def generate_chart(self):
        map_data = [[x["name"], x["value"]] for x in self.data]
        min_data, max_data = (
            min([d[1][0] for d in map_data]),
            max([d[1][0] for d in map_data]),
        )
        map_chart = (
            Map()
            .add(
                series_name="",
                data_pair=map_data,
                label_opts=opts.LabelOpts(is_show=False),
                is_map_symbol_show=False,
                itemstyle_opts={
                    "normal": {"areaColor": "#323c48", "borderColor": "#404a59"},
                    "emphasis": {
                        "label": {"show": True},
                        "areaColor": "rgba(255,255,255, 0.5)",
                    },
                },
            )
            .set_global_opts(
                title_opts=opts.TitleOpts(
                    title=self.title,
                    subtitle="数据来源：中国证券投资基金业协会",
                    pos_left="center",
                    pos_top="top",
                    title_textstyle_opts=opts.TextStyleOpts(
                        font_size=25, color="rgba(0,0,0, 0.9)"
                    ),
                ),
                tooltip_opts=opts.TooltipOpts(
                    is_show=True,
                    formatter=JsCode(
                        """function(params) {
                        if ('value' in params.data) {
                            return params.data.value[2] + ': ' + params.data.value[0];
                        }
                    }"""
                    ),
                ),
                visualmap_opts=opts.VisualMapOpts(
                    is_calculable=True,
                    dimension=0,
                    pos_left="10",
                    pos_top="center",
                    range_text=["High", "Low"],
                    range_color=["lightskyblue", "yellow", "orangered"],
                    textstyle_opts=opts.TextStyleOpts(color="#000"),
                    min_=min_data,
                    max_=max_data,
                ),
                toolbox_opts=opts.ToolboxOpts(is_show=False)
            )
        )

        bar_x_data = [x[0] for x in map_data]

        bar_y_data = [{"name": x[0], "value": x[1][0]} for x in map_data]
        bar = (
            Bar()
            .add_xaxis(xaxis_data=bar_x_data)
            .add_yaxis(
                series_name="",
                yaxis_index=1,
                y_axis=bar_y_data,
                label_opts=opts.LabelOpts(
                    is_show=True, position="right", formatter="{b}: {c}"
                ),
                category_gap='50%'
            )
            .reversal_axis()
            .set_global_opts(
                xaxis_opts=opts.AxisOpts(axislabel_opts=opts.LabelOpts(is_show=False)),
                yaxis_opts=opts.AxisOpts(axislabel_opts=opts.LabelOpts(is_show=False)),
                tooltip_opts=opts.TooltipOpts(is_show=False),
                visualmap_opts=opts.VisualMapOpts(
                    is_calculable=True,
                    dimension=0,
                    pos_left="10",
                    pos_top="center",
                    range_text=["High", "Low"],
                    range_color=["lightskyblue", "yellow", "orangered"],
                    textstyle_opts=opts.TextStyleOpts(color="#000"),
                    min_=min_data,
                    max_=max_data,
                )
            )
        )

        pie_data = [[x[0], x[1][0]] for x in map_data]
        percent_sum = sum([x[1][1] for x in map_data])
        rest_value = 0
        for d in map_data:
            rest_percent = 100.0
            rest_percent = rest_percent - percent_sum
            rest_value = d[1][0] * (rest_percent / d[1][1])
        pie_data.append(["其他省份", rest_value])
        pie = (
            Pie()
            .add(
                series_name="",
                data_pair=pie_data,
                radius=["12%", "20%"],

            )
        )

        grid_chart = (
            Grid(init_opts=opts.InitOpts(width="1366px", height="768px"))
            .add(
                bar,
                grid_opts=opts.GridOpts(
                    pos_left="5%", pos_right="55%", pos_top="70%", pos_bottom="5"

            title_tag.string = self.title
            soup.head.append(title_tag)

        # 将修改后的HTML写入文件
        with open(f"{self.title}.html", "w", encoding="utf-8") as file:
            file.write(soup.prettify())
